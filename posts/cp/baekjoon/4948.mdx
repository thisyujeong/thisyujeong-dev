```js
const fs = require('fs');
const input = fs.readFileSync('dev/stdin').toString().trim().split('\n').map(Number);

input.map((n) => {
  let count = 0;
  for (let i = n + 1; i <= 2 * n; i++) {
    let decimal = true;
    if (i <= 1) continue;
    for (let k = 2; k < i; k++) {
      if (i % k === 0) {
        decimal = false;
        break;
      }
    }
    if (decimal) count++;
  }
  if (n !== 0) console.log(count);
});
```
