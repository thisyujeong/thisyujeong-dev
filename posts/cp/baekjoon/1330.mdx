## Solution

```js
const fs = require('fs');
const [a, b] = fs.readFileSync('dev/stdin').toString().split(' ').map(Number);
function compare(a, b) {
  if (a > b) return console.log('>');
  if (a < b) return console.log('<');
  if (a == b) return console.log('==');
}
compare(a, b);
```
