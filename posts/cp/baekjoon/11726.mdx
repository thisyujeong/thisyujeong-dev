알고리즘: DP

```js
const fs = require('fs');
const filePath = process.platform === 'linux' ? 'dev/stdin' : '../input.txt';
const input = +fs.readFileSync(filePath).toString('utf-8');
const dp = [];
dp[1] = 1;
dp[2] = 2;
for (let i = 3; i <= input; i++) {
  dp[i] = (dp[i - 1] + dp[i - 2]) % 10007;
}
console.log(dp[input]);
```
