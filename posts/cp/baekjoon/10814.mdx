```js
const fs = require('fs');
const [N, ...input] = fs.readFileSync('../input.txt').toString().trim().split('\n');
const persons = input.map((person) => person.split(' '));
persons.sort((a, b) => Number(a[0]) - Number(b[0]));

let result = '';
persons.map((i) => (result += `${i.join(' ')}\n`));
console.log(result);
```
