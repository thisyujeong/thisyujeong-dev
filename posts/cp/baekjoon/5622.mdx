## Solution

```js
const input = require('fs').readFileSync('../input.txt').toString().trim().split('');
const dials = ['ABC', 'DEF', 'GHI', 'JKL', 'MNO', 'PQRS', 'TUV', 'WXYZ'];

let result = 0;
input.map((a) => {
  dials.map((dial, i) => {
    if (dial.includes(a)) result += 3 + i;
  });
});
console.log(result);
```
